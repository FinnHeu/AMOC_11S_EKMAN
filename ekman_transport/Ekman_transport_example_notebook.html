ds_ERA5 = xr.open_dataset(path_dataset)
print(ds_ERA5)

# set longitude grid to -180 180
ds_ERA5 = ds_ERA5.assign_coords(longitude=(((ds_ERA5.longitude + 180) % 360) - 180))

# choose ATL region
ds_ERA5 = ds_ERA5.sel(latitude=-11, longitude=np.arange(-37, 14, .25))

# Extract numpy arrays from netCDF (Ekman_Transport.py input is numpy arrays)
U_ERA5 = ds_ERA5.u.values
V_ERA5 = ds_ERA5.v.values
LAT_ERA5 = ds_ERA5.latitude.values * np.ones(np.shape(U_ERA5))

# Compute Ekman Tranpsort
U_ek_ERA5, V_ek_ERA5 = Ekman_Transport(U = U_ERA5, V = V_ERA5, LAT = LAT_ERA5, drag_coeff=5, extend_ranges=True, ekman_layer_constant=False)

# Zonally integrate Ekman velocity
v_ek_int_ERA5 = np.trapz(V_ek_ERA5,dx=27290,axis=1) / 1e6  #22790m between 0.25°longitude at 11°S
